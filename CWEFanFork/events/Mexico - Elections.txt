#1946 Pres Election #Revise to consider making the main party for pre-split PRI social_democrat
country_event = {
	id = 80095000
	title = "$YEAR$ Mexican Presidential Election"
	desc = "EVTDESC80095000"
	picture = "MEX_elections_historical"
	
	fire_only_once = yes

	trigger = {
		tag = MEX
		year = 1946
		NOT = { year = 1950 }
		
		OR = { government = presidential_dictatorship government = democracy government = democracy1 }
		is_vassal = no
		
		election = no
		
		#PRI parties in power
		OR = { ruling_party_ideology = socialist ruling_party_ideology = social_democrat ruling_party_ideology = big_tent ruling_party_ideology = conservative ruling_party_ideology = liberal }
	}

	mean_time_to_happen = { months = 3 }

	option = {
		name = "The PRI maintains its hegemony."
		
		random_owned = { #This should give the player or AI the modifier if they don't have a majority support
			limit = { owner = { AND = { government = democracy tag = MEX NOT = { OR = { pop_majority_ideology = big_tent pop_majority_ideology = socialist pop_majority_ideology = social_democrat } } } } }
			owner = { add_country_modifier = { name = revolution_chaos duration = 1825 } }
		}
		
		ruling_party_ideology = big_tent
		
		any_pop = { 
			ideology = { factor = 0.2 value = big_tent }
		}
		any_pop = { 
			limit = { strata = poor }
			ideology = { factor = 0.1 value = socialist }
		}
		
		#Opposition parties get angry
		any_pop = {
			scaled_militancy = {
				factor = 10
				ideology = traditionalist
			}
			scaled_militancy = {
				factor = 5
				ideology = conservative
			}
			scaled_militancy = {
				factor = 5
				ideology = socialist
			}
			scaled_militancy = {
				factor = 10
				ideology = communist
			}
			scaled_militancy = {
				factor = 10
				ideology = communist_social
			}
		}
		
		ai_chance = {
			factor = 0.9
			modifier = {
				factor = 0.6
				NOT = { OR = { pop_majority_ideology = big_tent pop_majority_ideology = socialist pop_majority_ideology = social_democrat } }
			}
			modifier = {
				factor = 2
				ruling_party_ideology = big_tent
			}
			modifier = {
				factor = 2
				AND = { #PRI dominance strenghtened by Soviet ties
					RUS = { exists = yes OR = { government = proletarian_dictatorship government = proletarian_dictatorship1 } }
					MEX = { in_sphere = RUS }
				}
			}
		}
	}

	option = {
		name = "Ezequiel Padilla Penaloza wins."
		ruling_party_ideology = traditionalist
		
		#USA gains some influence in Mexico
		random_country = {
			limit = { tag = USA is_greater_power = yes government = democracy OR = { ruling_party_ideology = traditionalist ruling_party_ideology = conservative ruling_party_ideology = big_tent } }
			relation = { who = THIS value = 50 }
			diplomatic_influence = { who = THIS value = 50 }
		}
		
		random_owned = { #This should give the player or AI the modifier if they don't have a majority support
			limit = { owner = { AND = { government = democracy tag = MEX NOT = { OR = { pop_majority_ideology = traditionalist pop_majority_ideology = conservative } } } } }
			owner = { add_country_modifier = { name = revolution_chaos duration = 1825 } }
		}
		
		any_pop = { 
			ideology = { factor = 0.15 value = conservative }
		}
		any_pop = { 
			limit = { strata = rich }
			ideology = { factor = 0.15 value = traditionalist }
		}
		
		#Liberals and Socialists are angry
		any_pop = {
			scaled_militancy = {
				factor = 10
				ideology = liberal
			}
			scaled_militancy = {
				factor = 5
				ideology = big_tent
			}
			scaled_militancy = {
				factor = 10
				ideology = progressive
			}
			scaled_militancy = {
				factor = 10
				ideology = socialist
			}
			scaled_militancy = {
				factor = 10
				ideology = communist
			}
			scaled_militancy = {
				factor = 10
				ideology = communist_social
			}
		}
	
		
		ai_chance = {
			factor = 0.1
			modifier = {
				factor = 6
				OR = { pop_majority_ideology = traditionalist pop_majority_ideology = conservative }
			}
			modifier = {
				factor = 4
				OR = { pop_majority_ideology = nationalist pop_majority_ideology = populist }
			}
			modifier = {
				factor = 2
				AND = { #PRI dominance weakened by American ties
					USA = { exists = yes NOT = { OR = { government = socialist_democracy government = proletarian_dictatorship government = proletarian_dictatorship1 } } }
					MEX = { in_sphere = USA }
				}
			}
		}
	}

}
